(() => {
  const API_BASE = '/space/api/exoplanets';
  const LOAD_QUERY = 'select top 10000 pl_name,hostname,discoverymethod,disc_year,disc_facility,disc_telescope,sy_snum,sy_pnum,pl_orbper,pl_orbsmax,pl_rade,pl_bmasse,pl_bmassprov,pl_eqt,pl_insol,pl_orbeccen,tran_flag,rv_flag,ima_flag from pscomppars order by pl_name asc';
  const DEFAULT_LIMIT = 50;
  const SORT_MAP = {
    disc_year_desc: (a,b) => (Number(b.disc_year)||0) - (Number(a.disc_year)||0) || String(a.pl_name||'').localeCompare(String(b.pl_name||'')),
    disc_year_asc: (a,b) => (Number(a.disc_year)||0) - (Number(b.disc_year)||0) || String(a.pl_name||'').localeCompare(String(b.pl_name||'')),
    planet_name: (a,b) => String(a.pl_name||'').localeCompare(String(b.pl_name||'')),
    host_name: (a,b) => String(a.hostname||'').localeCompare(String(b.hostname||'')) || String(a.pl_name||'').localeCompare(String(b.pl_name||'')),
    mass_desc: (a,b) => (Number(b.pl_bmasse)||-1) - (Number(a.pl_bmasse)||-1) || String(a.pl_name||'').localeCompare(String(b.pl_name||'')),
    radius_desc: (a,b) => (Number(b.pl_rade)||-1) - (Number(a.pl_rade)||-1) || String(a.pl_name||'').localeCompare(String(b.pl_name||'')),
    period_asc: (a,b) => (Number(a.pl_orbper)||Number.MAX_SAFE_INTEGER) - (Number(b.pl_orbper)||Number.MAX_SAFE_INTEGER) || String(a.pl_name||'').localeCompare(String(b.pl_name||''))
  };
  const TEMP_BINS = [0,100,200,300,400,600,800,1200,2000,5000];
  const el = {};
  const state = { allRows: [], filteredRows: [], visibleRows: [], scatterPoints: [], sourceSql: LOAD_QUERY };
  function byId(id){ return document.getElementById(id); }
  function apiUrl(sql, format='json'){ return `${API_BASE}?query=${encodeURIComponent(sql)}&format=${encodeURIComponent(format)}`; }
  async function fetchJson(sql){ const response = await fetch(apiUrl(sql,'json'), { method:'GET', credentials:'same-origin', headers:{Accept:'application/json'} }); if(!response.ok) throw new Error(`Proxy request failed (${response.status})`); return response.json(); }
  function formatNumber(value, digits=2){ if(value===null || value===undefined || value==='') return '—'; const num=Number(value); if(Number.isNaN(num)) return String(value); return num.toLocaleString(undefined,{maximumFractionDigits:digits}); }
  function setStatus(message, kind='info'){ if(!el.status) return; el.status.textContent = message; el.status.dataset.kind = kind; }
  function setMetric(id, value){ const target = byId(id); if(target) target.textContent = value; }
  function parseNumber(value){ const n = Number(value); return Number.isFinite(n) ? n : null; }
  function parseRange(idMin, idMax){ return { min: parseNumber(el[idMin]?.value || ''), max: parseNumber(el[idMax]?.value || '') }; }
  function normalizeText(value){ return String(value || '').toLowerCase().replace(/[^a-z0-9]+/g, ''); }
  function searchMatches(row, searchValue){
    const term = normalizeText(searchValue); if(!term) return true;
    const planet = normalizeText(row.pl_name); const host = normalizeText(row.hostname);
    return planet.includes(term) || host.includes(term);
  }
  function methodMatches(row, methodValue){
    const term = String(methodValue || '').trim().toLowerCase(); if(!term) return true;
    return String(row.discoverymethod || '').toLowerCase().includes(term);
  }
  function rangeMatches(value, range){
    const num = parseNumber(value);
    if(range.min === null && range.max === null) return true;
    if(num === null) return false;
    if(range.min !== null && num < range.min) return false;
    if(range.max !== null && num > range.max) return false;
    return true;
  }
  function currentComparator(){ return SORT_MAP[el.sort?.value || 'disc_year_desc'] || SORT_MAP.disc_year_desc; }
  function currentLimit(){ const parsed = Number(el.limit?.value || DEFAULT_LIMIT); return Number.isFinite(parsed) && parsed > 0 ? parsed : DEFAULT_LIMIT; }
  function applyFilters(){
    const searchValue = el.search?.value || '';
    const methodValue = el.discoveryMethod?.value || '';
    const yearRange = parseRange('yearFrom','yearTo');
    const radiusRange = parseRange('radiusMin','radiusMax');
    const eqtRange = parseRange('eqtMin','eqtMax');
    const insolRange = parseRange('insolMin','insolMax');
    state.filteredRows = state.allRows.filter((row) => {
      return searchMatches(row, searchValue)
        && methodMatches(row, methodValue)
        && rangeMatches(row.disc_year, yearRange)
        && rangeMatches(row.pl_rade, radiusRange)
        && rangeMatches(row.pl_eqt, eqtRange)
        && rangeMatches(row.pl_insol, insolRange);
    });
    state.filteredRows.sort(currentComparator());
    state.visibleRows = state.filteredRows.slice(0, currentLimit());
    renderVisibleRows();
    setStatus(`Loaded ${state.visibleRows.length.toLocaleString()} visible row(s) from ${state.filteredRows.length.toLocaleString()} filtered match(es).`, 'success');
  }
  function clearCanvas(canvas){ if(!canvas) return null; const ctx=canvas.getContext('2d'); ctx.clearRect(0,0,canvas.width,canvas.height); ctx.fillStyle='#061008'; ctx.fillRect(0,0,canvas.width,canvas.height); return ctx; }
  function drawEmptyState(canvas,message){ const ctx=clearCanvas(canvas); if(!ctx) return; ctx.fillStyle='rgba(131,255,163,.85)'; ctx.font='16px sans-serif'; ctx.textAlign='center'; ctx.textBaseline='middle'; ctx.fillText(message, canvas.width/2, canvas.height/2); }
  function drawAxes(ctx,width,height,margin,maxVal){ const plotH=height-margin.top-margin.bottom; ctx.strokeStyle='rgba(131,255,163,.22)'; ctx.lineWidth=1; for(let step=0; step<=4; step+=1){ const y=margin.top+(plotH/4)*step; ctx.beginPath(); ctx.moveTo(margin.left,y); ctx.lineTo(width-margin.right,y); ctx.stroke(); } ctx.strokeStyle='rgba(131,255,163,.4)'; ctx.beginPath(); ctx.moveTo(margin.left,margin.top); ctx.lineTo(margin.left,height-margin.bottom); ctx.lineTo(width-margin.right,height-margin.bottom); ctx.stroke(); ctx.fillStyle='rgba(131,255,163,.88)'; ctx.font='12px sans-serif'; ctx.textAlign='right'; for(let step=0; step<=4; step+=1){ const value=Math.round(maxVal-(maxVal/4)*step); const y=margin.top+(plotH/4)*step; ctx.fillText(String(value), margin.left-8, y+4); } }
  function drawBarChart(canvas,entries,options={}){ if(!canvas) return; if(!entries.length) return drawEmptyState(canvas, options.emptyMessage || 'No chart data for current results'); const ctx=clearCanvas(canvas); const {width,height}=canvas; const margin={top:28,right:18,bottom:52,left:54}; const plotW=width-margin.left-margin.right; const plotH=height-margin.top-margin.bottom; const maxVal=Math.max(...entries.map(([,v])=>v),1); const barWidth=plotW/entries.length*0.72; const gap=plotW/entries.length*0.28; drawAxes(ctx,width,height,margin,maxVal); entries.forEach(([label,value],index)=>{ const x=margin.left+index*(barWidth+gap)+gap/2; const barHeight=(value/maxVal)*(plotH-4); const y=height-margin.bottom-barHeight; ctx.fillStyle=options.barColor || 'rgba(131,255,163,.82)'; ctx.fillRect(x,y,barWidth,barHeight); ctx.fillStyle='rgba(131,255,163,.96)'; ctx.textAlign='center'; ctx.fillText(String(value), x+barWidth/2, y-6); ctx.save(); ctx.translate(x+barWidth/2, height-margin.bottom+8); ctx.rotate(-Math.PI/5); ctx.font='12px sans-serif'; ctx.fillStyle='rgba(131,255,163,.82)'; const shortLabel=label.length>14 ? `${label.slice(0,12)}…` : label; ctx.fillText(shortLabel,0,0); ctx.restore(); }); }
  function drawYearChart(canvas,entries){ if(!canvas) return; if(!entries.length) return drawEmptyState(canvas,'No year data for current results'); const ctx=clearCanvas(canvas); const {width,height}=canvas; const margin={top:28,right:18,bottom:44,left:54}; const plotW=width-margin.left-margin.right; const plotH=height-margin.top-margin.bottom; const maxVal=Math.max(...entries.map(([,v])=>v),1); drawAxes(ctx,width,height,margin,maxVal); ctx.strokeStyle='rgba(131,255,163,.44)'; ctx.beginPath(); entries.forEach(([year,count],index)=>{ const x=margin.left+(plotW/Math.max(entries.length-1,1))*index; const y=height-margin.bottom-(count/maxVal)*(plotH-8); if(index===0) ctx.moveTo(x,y); else ctx.lineTo(x,y); }); ctx.stroke(); entries.forEach(([year,count],index)=>{ const x=margin.left+(plotW/Math.max(entries.length-1,1))*index; const y=height-margin.bottom-(count/maxVal)*(plotH-8); ctx.fillStyle='rgba(131,255,163,.95)'; ctx.beginPath(); ctx.arc(x,y,4.5,0,Math.PI*2); ctx.fill(); ctx.font='12px sans-serif'; ctx.textAlign='center'; ctx.fillText(String(year), x, height-margin.bottom+18); ctx.fillText(String(count), x, y-10); }); }
  function buildMethodEntries(rows){ const counts=new Map(); rows.forEach((row)=>{ const key=row.discoverymethod || 'Unknown'; counts.set(key,(counts.get(key)||0)+1); }); return [...counts.entries()].sort((a,b)=>b[1]-a[1]).slice(0,8); }
  function buildYearEntries(rows){ const counts=new Map(); rows.forEach((row)=>{ const year=parseNumber(row.disc_year); if(year!==null) counts.set(year,(counts.get(year)||0)+1); }); return [...counts.entries()].sort((a,b)=>a[0]-b[0]); }
  function buildTopEntries(rows,keyName){ const counts=new Map(); rows.forEach((row)=>{ const key=row[keyName] || 'Unknown'; counts.set(key,(counts.get(key)||0)+1); }); return [...counts.entries()].sort((a,b)=>b[1]-a[1]).slice(0,8); }
  function buildTempEntries(rows){ const counts=Array(TEMP_BINS.length-1).fill(0); rows.forEach((row)=>{ const v=parseNumber(row.pl_eqt); if(v===null || v<0) return; for(let i=0;i<TEMP_BINS.length-1;i+=1){ if(v>=TEMP_BINS[i] && v<TEMP_BINS[i+1]){ counts[i]+=1; return; } } if(v>=TEMP_BINS[TEMP_BINS.length-1]) counts[counts.length-1]+=1; }); return counts.map((count,i)=>[`${TEMP_BINS[i]}-${TEMP_BINS[i+1]}K`, count]).filter(([,count])=>count>0); }
  function buildScatterPoints(rows){ return rows.map((row)=>({ label:row.pl_name, host:row.hostname, mass:parseNumber(row.pl_bmasse), radius:parseNumber(row.pl_rade), year:row.disc_year, method:row.discoverymethod })).filter((p)=>p.mass!==null && p.mass>0 && p.radius!==null && p.radius>0).slice(0,250); }
  function hideScatterTooltip(){ if(el.scatterTooltip) el.scatterTooltip.hidden=true; }
  function showScatterTooltip(point,event){ if(!el.scatterTooltip || !el.scatterContainer) return; el.scatterTooltip.innerHTML=`<strong>${point.label || 'Unknown planet'}</strong><div>Host: ${point.host || '—'}</div><div>Mass: ${formatNumber(point.mass,3)} M⊕</div><div>Radius: ${formatNumber(point.radius,3)} R⊕</div><div>Year: ${point.year ?? '—'}</div><div>Method: ${point.method || '—'}</div>`; const rect=el.scatterContainer.getBoundingClientRect(); const x=event.clientX-rect.left+12; const y=event.clientY-rect.top+12; el.scatterTooltip.style.left=`${x}px`; el.scatterTooltip.style.top=`${y}px`; el.scatterTooltip.hidden=false; }
  function drawScatterChart(canvas,points){ if(!canvas) return; if(!points.length){ state.scatterPoints=[]; hideScatterTooltip(); return drawEmptyState(canvas,'Need both mass and radius values to draw scatter plot'); } const ctx=clearCanvas(canvas); const {width,height}=canvas; const margin={top:24,right:22,bottom:58,left:68}; const plotW=width-margin.left-margin.right; const plotH=height-margin.top-margin.bottom; const maxX=Math.max(...points.map((p)=>p.mass),1); const maxY=Math.max(...points.map((p)=>p.radius),1); ctx.strokeStyle='rgba(131,255,163,.22)'; for(let step=0; step<=4; step+=1){ const gx=margin.left+(plotW/4)*step; const gy=margin.top+(plotH/4)*step; ctx.beginPath(); ctx.moveTo(gx,margin.top); ctx.lineTo(gx,height-margin.bottom); ctx.stroke(); ctx.beginPath(); ctx.moveTo(margin.left,gy); ctx.lineTo(width-margin.right,gy); ctx.stroke(); } ctx.strokeStyle='rgba(131,255,163,.45)'; ctx.beginPath(); ctx.moveTo(margin.left,margin.top); ctx.lineTo(margin.left,height-margin.bottom); ctx.lineTo(width-margin.right,height-margin.bottom); ctx.stroke(); ctx.fillStyle='rgba(131,255,163,.85)'; ctx.font='12px sans-serif'; ctx.textAlign='center'; for(let step=0; step<=4; step+=1){ const xValue=(maxX/4)*step; const yValue=maxY-(maxY/4)*step; const gx=margin.left+(plotW/4)*step; const gy=margin.top+(plotH/4)*step; ctx.fillText(xValue.toFixed(step===0?0:1), gx, height-margin.bottom+18); ctx.textAlign='right'; ctx.fillText(yValue.toFixed(step===4?0:1), margin.left-8, gy+4); ctx.textAlign='center'; } ctx.save(); ctx.translate(width/2,height-16); ctx.fillText('Mass (Earth masses)',0,0); ctx.restore(); ctx.save(); ctx.translate(18,height/2); ctx.rotate(-Math.PI/2); ctx.fillText('Radius (Earth radii)',0,0); ctx.restore(); state.scatterPoints=points.map((point)=>{ const x=margin.left+(point.mass/maxX)*plotW; const y=height-margin.bottom-(point.radius/maxY)*plotH; ctx.fillStyle='rgba(131,255,163,.78)'; ctx.beginPath(); ctx.arc(x,y,4.2,0,Math.PI*2); ctx.fill(); return { ...point, x, y, r: 7 }; }); }
  function bindScatterHover(){ if(!el.scatterChart) return; el.scatterChart.addEventListener('mousemove', (event)=>{ if(!state.scatterPoints.length){ hideScatterTooltip(); return; } const rect=el.scatterChart.getBoundingClientRect(); const scaleX=el.scatterChart.width/rect.width; const scaleY=el.scatterChart.height/rect.height; const x=(event.clientX-rect.left)*scaleX; const y=(event.clientY-rect.top)*scaleY; const match=state.scatterPoints.find((point)=>{ const dx=x-point.x; const dy=y-point.y; return Math.sqrt(dx*dx+dy*dy) <= point.r; }); if(match) showScatterTooltip(match,event); else hideScatterTooltip(); }); el.scatterChart.addEventListener('mouseleave', hideScatterTooltip); }
  function renderCharts(rows){ drawBarChart(el.methodChart, buildMethodEntries(rows), { emptyMessage:'No discovery method data for current results' }); drawYearChart(el.yearChart, buildYearEntries(rows)); drawBarChart(el.tempChart, buildTempEntries(rows), { emptyMessage:'No temperature data for current results', barColor:'rgba(255,180,90,.82)' }); drawBarChart(el.facilityChart, buildTopEntries(rows,'disc_facility'), { emptyMessage:'No facility data for current results', barColor:'rgba(120,200,255,.82)' }); drawBarChart(el.telescopeChart, buildTopEntries(rows,'disc_telescope'), { emptyMessage:'No telescope data for current results', barColor:'rgba(200,170,255,.82)' }); drawScatterChart(el.scatterChart, buildScatterPoints(rows)); }
  function renderVisibleRows(){ renderTable(state.visibleRows); setMetric('exoLoadedRows', formatNumber(state.visibleRows.length,0)); }
  function renderTable(rows){ if(!el.resultsBody) return; if(!rows.length){ el.resultsBody.innerHTML='<tr><td colspan="9" class="exo-empty">No exoplanets matched the current filters.</td></tr>'; el.resultCount.textContent='0'; renderDetails(null); renderCharts([]); return; } el.resultCount.textContent=rows.length.toLocaleString(); el.resultsBody.innerHTML=rows.map((row,index)=>`\n<tr data-index="${index}"><td>${row.pl_name || '—'}</td><td>${row.hostname || '—'}</td><td>${row.disc_year ?? '—'}</td><td>${row.discoverymethod || '—'}</td><td>${formatNumber(row.pl_bmasse,2)}</td><td>${formatNumber(row.pl_rade,2)}</td><td>${formatNumber(row.pl_orbper,3)}</td><td>${formatNumber(row.pl_eqt,0)}</td><td>${row.disc_facility || '—'}</td></tr>`).join(''); renderDetails(rows[0]); renderCharts(rows); const firstRow=el.resultsBody.querySelector('tr[data-index="0"]'); if(firstRow) firstRow.classList.add('is-active'); }
  function renderDetails(row){ if(!el.details) return; if(!row){ el.details.innerHTML='<p class="exo-empty">Select an exoplanet row to inspect detailed values.</p>'; return; } const items=[['Planet Name',row.pl_name],['Host Star',row.hostname],['Discovery Method',row.discoverymethod],['Discovery Year',row.disc_year],['Discovery Facility',row.disc_facility],['Discovery Telescope',row.disc_telescope],['Stars in System',row.sy_snum],['Planets in System',row.sy_pnum],['Orbital Period (days)',formatNumber(row.pl_orbper,5)],['Semi-major Axis (AU)',formatNumber(row.pl_orbsmax,5)],['Radius (Earth radii)',formatNumber(row.pl_rade,4)],['Best Mass Estimate (Earth masses)',formatNumber(row.pl_bmasse,4)],['Mass Provenance',row.pl_bmassprov],['Equilibrium Temp (K)',formatNumber(row.pl_eqt,0)],['Insolation (Earth flux)',formatNumber(row.pl_insol,3)],['Eccentricity',formatNumber(row.pl_orbeccen,4)],['Transit Detection Flag',row.tran_flag],['Radial Velocity Flag',row.rv_flag],['Imaging Flag',row.ima_flag]]; el.details.innerHTML=`<dl class="exo-detail-grid">${items.map(([label,value])=>`\n<div class="exo-detail-item"><dt>${label}</dt><dd>${value===null || value===undefined || value==='' ? '—' : value}</dd></div>`).join('')}</dl>`; }
  function buildCsv(rows){ const cols=['pl_name','hostname','discoverymethod','disc_year','disc_facility','disc_telescope','pl_bmasse','pl_rade','pl_orbper','pl_eqt','pl_insol']; const esc=(v)=>`"${String(v ??   function downloadBlob(name, content, type){ const blob=new Blob([content], {type}); const url=URL.createObjectURL(blob); const a=document.createElement('a'); a.href=url; a.download=name; document.body.appendChild(a); a.click(); a.remove(); setTimeout(()=>URL.revokeObjectURL(url), 1000); }

  async function loadSummary(){ try{ const [countRows,transitRows,yearRows]=await Promise.all([ fetchJson('select count(*) as total_planets from pscomppars'), fetchJson('select count(*) as transit_planets from pscomppars where tran_flag = 1'), fetchJson('select max(disc_year) as latest_discovery_year from pscomppars') ]); setMetric('exoTotalPlanets', formatNumber(countRows?.[0]?.total_planets,0)); setMetric('exoTransitPlanets', formatNumber(transitRows?.[0]?.transit_planets,0)); setMetric('exoLatestDiscoveryYear', formatNumber(yearRows?.[0]?.latest_discovery_year,0)); }catch(error){ setMetric('exoTotalPlanets','Error'); setMetric('exoTransitPlanets','Error'); setMetric('exoLatestDiscoveryYear','Error'); setStatus(error.message,'error'); } }

Filters applied locally.`, 'success'); }catch(error){ state.allRows=[]; state.filteredRows=[]; state.visibleRows=[]; renderTable([]); setStatus(error.message,'error'); } }
